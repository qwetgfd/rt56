CREATE OR ALTER PROCEDURE dbo.sel_flpSharePointSourceByConfigId
    @flpConfigurationId NVARCHAR(100)
AS
BEGIN
    SET NOCOUNT ON;

    SELECT
        s.flpConfigurationId,
        s.sharePointApplicationId,
        s.sharePointApplicationSiteId,
        s.sharePointLibraryName,
        s.sharePointFolderPath,
        a.displayname AS sharePointApplicationName,
        site.sitename AS sharePointSiteName
    FROM dbo.di_flpSharePointSource s
    LEFT JOIN dbo.di_application a ON a.applicationid = s.sharePointApplicationId
    LEFT JOIN dbo.di_applicationsite site ON site.applicationsiteid = s.sharePointApplicationSiteId
    WHERE s.flpConfigurationId = @flpConfigurationId
      AND s.active = 1;
END;
GO
