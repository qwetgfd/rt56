-- SharePoint-only table: di_sharepointlogs
-- Safe to backup, drop, recreate. Replace backup18June2026 with your date.

SET NOCOUNT ON;
GO

IF OBJECT_ID(N'[dbo].[di_sharepointlogs]', N'U') IS NOT NULL
BEGIN
    IF OBJECT_ID(N'[dbo].[di_sharepointlogs_backup18June2026]', N'U') IS NOT NULL
        DROP TABLE [dbo].[di_sharepointlogs_backup18June2026];
    EXEC sp_rename N'dbo.di_sharepointlogs', N'di_sharepointlogs_backup18June2026', N'OBJECT';
END
GO

CREATE TABLE [dbo].[di_sharepointlogs] (
    [sharepointlogid] BIGINT IDENTITY(1,1) NOT NULL,
    [loglevel] NVARCHAR(16) NOT NULL,
    [logger] NVARCHAR(256) NOT NULL,
    [message] NVARCHAR(MAX) NOT NULL,
    [exception] NVARCHAR(MAX),
    [applicationid] UNIQUEIDENTIFIER,
    [applicationname] NVARCHAR(128),
    [usedbyupn] NVARCHAR(256),
    [usedbydisplayname] NVARCHAR(256),
    [requestmethod] NVARCHAR(16),
    [requestpath] NVARCHAR(512),
    [createdon] DATETIME2(3) NOT NULL DEFAULT (getutcdate()),
    CONSTRAINT [PK_di_sharepointlogs] PRIMARY KEY CLUSTERED ([sharepointlogid])
);
CREATE NONCLUSTERED INDEX [ix_di_sharepointlogs_applicationid] ON [dbo].[di_sharepointlogs] ([applicationid]);
CREATE NONCLUSTERED INDEX [ix_di_sharepointlogs_createdon] ON [dbo].[di_sharepointlogs] ([createdon]);
