/* SharePoint Configuration First – READ-ONLY diagnostics (prod) */

SET NOCOUNT ON;

DECLARE @Id1 NVARCHAR(100) = '029B59E4-B249-47B3-B71D-662DF964E2E7';  -- DI_APNORAC14
DECLARE @Id2 NVARCHAR(100) = 'F782404F-D85D-482F-A623-2C7451443CEE';  -- second row from list

-- 1) What GetProcessList / Function App gets
EXEC dbo.sel_flpConfigurationList @processTypeId = 6;

-- 2) Main table SharePoint columns
SELECT
    flpConfigurationId,
    process_name,
    locationTypeId,
    processTypeId,
    is_active,
    sourcePath,
    sharePointApplicationId,
    sharePointApplicationSiteId,
    sharePointLibraryName,
    sharePointFolderPath
FROM dbo.di_flpConfiguration
WHERE flpConfigurationId IN (@Id1, @Id2);

-- 3) Side table (your screenshot showed NO rows here)
SELECT *
FROM dbo.di_flpSharePointSource
WHERE flpConfigurationId IN (@Id1, @Id2);

-- 4) Side table read SP
EXEC dbo.sel_flpSharePointSourceByConfigId @flpConfigurationId = @Id1;
EXEC dbo.sel_flpSharePointSourceByConfigId @flpConfigurationId = @Id2;

-- 5) Edit load SP (already joins side table in repo version)
EXEC dbo.sel_processConfigurationById @flpConfigurationId = @Id1, @tabName = NULL;

-- 6) All SharePoint configs: main vs side table in one view
SELECT
    f.flpConfigurationId,
    f.process_name,
    f.sourcePath,
    f.sharePointApplicationId     AS main_appId,
    f.sharePointApplicationSiteId AS main_siteId,
    f.sharePointLibraryName       AS main_library,
    f.sharePointFolderPath        AS main_folder,
    s.sharePointApplicationId     AS side_appId,
    s.sharePointApplicationSiteId AS side_siteId,
    s.sharePointLibraryName       AS side_library,
    s.sharePointFolderPath        AS side_folder
FROM dbo.di_flpConfiguration f
LEFT JOIN dbo.di_flpSharePointSource s
    ON s.flpConfigurationId = f.flpConfigurationId AND s.active = 1
WHERE f.is_active = 1
  AND f.processTypeId = 6
  AND f.locationTypeId = 4
ORDER BY f.process_name;
